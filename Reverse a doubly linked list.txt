static DoublyLinkedListNode reverse(DoublyLinkedListNode head) {

        DoublyLinkedListNode start=head;
        DoublyLinkedListNode end=head;
        int temp;
        int k=1,i=0;
        while(end.next!=null){
            k++;
            end=end.next;
        }

        while(i<k){
            temp=start.data;
            start.data=end.data;
            end.data=temp;

            end=end.prev;
            start=start.next;
            i++;
            k--;
        }
        return head;

    }
