#include<stdio.h>
int main()
{
    int n,a[100];
    //printf("enter the number of students::")
    scanf("%d\n",&n);
    //printf("enter the marks");
    for(int i=0;i<n;i++)
    {
        scanf("%d\n",&a[i]);
    }
    for(int i=0;i<n;i++)
    {
        if(a[i]<38||a[i]%5<3)
            printf("%d\n",a[i]);
        else
        {
            a[i]=a[i]+(5-(a[i]%5));
            printf("%d\n",a[i]);
        }
    }
}
