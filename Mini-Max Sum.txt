int main()
{
    long long int i,j,Smin=0,Smax=0,c;
   
   long long int a[5];
    for(i=0;i<5;i++)
        scanf("%lld",&a[i]);
    for(i=0;i<4;i++)
    {
        for(j=0;j<4-i;j++)
        {
            if(a[j]>a[j+1])
            {
                c=a[j];
                a[j]=a[j+1];
                a[j+1]=c;
            }
        }
    }
 
    for(i=1;i<5;i++)
        Smax=Smax+a[i];
   
    for(i=0;i<4;i++)
        Smin=Smin+a[i];
    printf("%lld %lld",Smin,Smax);
    return 0;
}
