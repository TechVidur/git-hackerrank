#include<stdio.h>
int main()
{
long long int n,i,j,cand=0,temp,c,key;
scanf("%lld",&n);    
long long int a[n];
for(i=0;i<n;i++)
    scanf("%lld",&a[i]);
   for (i = 1; i < n; i++)
   {
       key = a[i];
       j = i-1;
 
       /* Move elements of arr[0..i-1], that are
          greater than key, to one position ahead
          of their current position */
       while (j >= 0 && a[j] > key)
       {
           a[j+1] = a[j];
           j = j-1;
       }
       a[j+1] = key;
   }

temp=a[n-1];
for(j=n-1;j>=0;j--)
{
    if(temp==a[j])
        cand++;
}
printf("%lld",cand);
return 0;
}
