#include<stdio.h>
#include<stdlib.h>
int main()
{
    int n,i,j;
    int sum1=0,sum2=0,d;
    scanf("%d",&n);
    int a[n][n];
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(i==j)
                sum1=sum1+a[i][j];
            if(i==n-j-1)
                sum2=sum2+a[i][j];
        }
    }
    
    d=abs(sum1-sum2);
    printf("%d",d);
    return 0;
}
